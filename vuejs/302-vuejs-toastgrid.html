# vuejs 에 toastgrid 사용

Vue.js 에서 사용 가능한 Grid 를 찾다가 NHN 에서 만든 Toast Grid 가 있어 설치 방법 및 사용법에 대해 기록을 남깁니다. 보다 자세한 내용은 공식 사이트를 참고 하시기 바랍니다. https://github.com/nhn/tui.grid/tree/master/packages/toast-ui.vue-grid (opens new window)

# 1. 설치 및 설정

vuejs 에 toastgrid 설치 및 설정 하는 방법입니다.

# 설치

yarn 으로 설치 할때는 아래와 같이 입력합니다.

yarn add @toast-ui/vue-grid
1

npm 설치 방법은 아래와 같습니다.

npm install --save @toast-ui/vue-grid
1

# 설정

grid 컴포넌트를 전역에서 사용할수 있도록 main.js 에서 아래와 같이 입력합니다. tui-date-picker.css 와 tui-time-picker.css 는 grid 에서 날짜 및 시간을 선택할때 사용됩니다. 페이징 처리를 위한 tui-pagination.css 파일도 import 합니다.

import 'tui-grid/dist/tui-grid.css'
import 'tui-date-picker/dist/tui-date-picker.css';
import 'tui-time-picker/dist/tui-time-picker.css';
import 'tui-pagination/dist/tui-pagination.css';
import { Grid } from '@toast-ui/vue-grid';
Vue.component('grid',Grid)

# 4.기본 그리드

기본 그리드 입니다.

columns 부분에 칼럼 헤더에 표시될 header 와 실제 값을 출력할 name 속성을 정해놓고 data에는 json 형태의 값을 연결해 주면 위와 같이 그리드에 데이터가 표시됩니다. 여기서 중요한 부분은 2번라인의 v-if gridData.length > 0 보다 큰경우 즉 그리드에 데이터가 있을 경우만 표시하도록 하는 부분인데요 이부분이 없으면 원하는 되로 동작하지 않습니다. v-if문의 빼고도 실행해 보시기 바랍니다.


 
























































<template>
  <grid v-if="gridData.length > 0"
        ref="tuiGrid2"
        :data="gridData"
        :columns="columns"
        :scrollX="scrollX"
        :scrollY="scrollY"
    ></grid>
</template>

<script>
export default {
  inheritAttrs: false,
  name: "Exam3",
  mounted() {
    this.getData();
  },
  data() {
    return {
      gridData: [],
      columns: [
        {
          header: '이름',
          name: 'name'
        },
        {
          header: '아티스트',
          name: 'artist'
        },
        {
          header: '형태',
          name: 'type'
        },
        {
          header: '출시일',
          name: 'release'
        },
        {
          header: '장르',
          name: 'genre'
        }
      ],
      scrollX: false,
      scrollY: false,
    }
  },
  methods: {
    getData() {
      this.$http.get('/json.json')
          .then((res) => {
            console.log(res.data);
            this.gridData = res.data;
          })
    },
  }
}
</script>
