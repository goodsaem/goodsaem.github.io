<template>
  <div>
    <input type="button" value="test" @click="randColor" />
    <v-stage
        :config="{
          width : 1024,
          height : 1024
        }"
        ref="stage">
      <v-layer ref="layer">
        <v-rect v-for="(rect,index) in rectConfig"  :keys="'rect_' + index"
          :ref="'rect_' + index"
          :config="rect">
        </v-rect>
      </v-layer>
    </v-stage>
  </div>
</template>

<script>
import Konva from 'konva';
export default {
  name: "MorTest",
  beforeCreate() {
  },
  created() {
    for(let i=1; i<=100; i++) {
      for(let j=1; j<=100; j++) {
        this.rectConfig.push({
          x : (j*9),
          y : (i*9),
          width : 7,
          height: 7,
          fill: 'rgb(255,255,255)',
          stroke: 'rgb(150,150,150)',
          strokeWidth:1,
        });
      }
    }
  },
  mounted() {
  },
  data() {
    return {
      rectConfig: [],
    }
  },
  methods: {
    randColor() {
      for(let i=0; i<=9999; i++) {
        const color = Konva.Util.getRandomColor();
        this.rectConfig[i].fill = color;
      }
    }
  }
}
</script>
