# 3. 조건문

정말로 많이 사용하는 문장입니다. 조건을 만족했을때 하위 문장을 실행합니다.

# 1. if else-if else

# 문법

if(조건식1){
    조건식1이 참(true)일때 실행되는 문장
}else if(조건식2){
    조건식2이 참(true)일때 실행되는 문장
}else{
    위에 조건식을 만족하지 않을때 실행되는 문장
}

# 예제

text box에 1~100의 값을 넣으면 성적을 출력하는 간단한 예제를 만들어 보겠습니다.

 - 결과1 : F등급
- 결과2 : F등급
- 결과3 : F등급

parseInt 함수는 문자를 숫자로 바꾸는 함수이구요 입력된 값에 따라 result 함수에서 판정하여 결과를 리턴하는 간단한 문장인데요 여기서 유의해야 할 부분은 if문은 위에서 부터 아래로 실행 된다는 부분입니다. 중간에 조건을 만족하는 문장이 나오면 더이상 문장을 수행하지 않습니다. 95점 이상이면 A+ 라고 출력하기 위해 result2 번 함수와 같이 90점 밑에 해당 문장을 넣으면 95점 이상 조건을 만나기도 전에 90점 이상이라는 조건을 만족하여 A라고 의도하지 않은 결과가 나옵니다. result3 함수와 같이 위쪽으로 조건문을 올리면 생각한되로 동작합니다.

<template>
  <div>
    <span>
       <el-input placeholder="점수를 입력하세요" v-model="score" />
    </span>
    <span>
      - 결과1 : {{ result }}  <br/>
      - 결과2 : {{ result2 }} <br/>
      - 결과3 : {{ result3 }} <br/>
    </span>
  </div>
</template>

<script>
export default {
  name: "Exam6",
  created() {},
  data() {
    return {
      score : 0,
    }
  },
  computed: {
    result() {
      if(parseInt(this.score) >= 90) {
        return 'A등급';
      } else if(parseInt(this.score) >= 80) {
        return 'B등급';
      } else if(parseInt(this.score) >= 70) {
        return 'C등급';
      } else if(parseInt(this.score) >= 60) {
        return 'D등급';
      } else {
        return 'F등급';
      }
    },
    result2() {
      if(parseInt(this.score) >= 90) {
        return 'A등급';
      } else if(parseInt(this.score) >= 95) {
        return 'A+등급';
      } else if(parseInt(this.score) >= 80) {
        return 'B등급';
      } else if(parseInt(this.score) >= 70) {
        return 'C등급';
      } else if(parseInt(this.score) >= 60) {
        return 'D등급';
      } else {
        return 'F등급';
      }
    },
    result3() {
      if(parseInt(this.score) >= 95) {
        return 'A+등급';
      } else if(parseInt(this.score) >= 90) {
        return 'A등급';
      } else if(parseInt(this.score) >= 80) {
        return 'B등급';
      } else if(parseInt(this.score) >= 70) {
        return 'C등급';
      } else if(parseInt(this.score) >= 60) {
        return 'D등급';
      } else {
        return 'F등급';
      }
    }
  }
}
</script>

# 2. ?

# 문법

let 변수 = 조건 ? true인경우 : false 인경우
1

? 물음표 연산자 라구요 하구요 피 연산자가 3개 (=,?,: ) 라고 해서 삼항 연산자라고도 하는데 조건이 참인경우에는 : 앞에 문장을 거짓인 경우에는 뒤에 문장을 변수에 대입을 합니다. 위에 성적을 나타내는 문장을 아래와 같이 표현해 보았습니다.

 - 결과 : F등급

코드는 짧아졌지만 가독성이 좋지 않습니다. 한참을 생각해야 되는 코드는 좋은 코드가 아니라고 생각하기위 3항 연산으로만 사용하는걸 추천드립니다.

<template>
  <div>
    <span>
       <el-input placeholder="점수를 입력하세요" v-model="score" />
    </span>
    <span>
      - 결과 : {{ result }}
    </span>
  </div>
</template>

<script>
export default {
  name: "Exam7",
  created() {},
  data() {
    return {
      score : 0,
    }
  },
  computed: {
    result() {
      return this.score >= 95 ? 'A+등급' : this.score >= 90 ? 'A등급' : this.score >= 80 ? 'B등급' :
             this.score >= 70 ? 'C등급'  : this.score >= 60 ? 'D등급' : 'F등급'
    }
  }
}
</script>
